# ESP32OS UI Toolkit - runtime dependencies (trimmed)

# Core (required)
pygame>=2.5.2,<3

# JSON Schema validation (required): tools/validate_design.py enforces
# schemas/ui_design.schema.json as a hard release gate for codegen, demo
# generation, visual verification, and PlatformIO builds. Schema validation
# is mandatory and is NOT silently skipped when this dependency is absent.
jsonschema>=4.20.0,<5

# Optional (USB/serial live preview)
pyserial>=3.5,<4

# Embedded toolchain (required for the integrated build/flash workflow):
# tools/build.py and the designer's Build/Flash modal invoke
# `python -m platformio run` to compile/upload real firmware. Bundling
# PlatformIO as a managed dependency makes "the compiler is part of the app"
# literally true — no separate manual toolchain install. tools/build.py
# `ensure_platformio()` verifies it is importable and prints an actionable
# `pip install -r requirements.txt` hint if it is missing.
platformio>=6.1,<7
# Required by the ESP-IDF build step (esptool/elf2image). PlatformIO's penv
# carries it, but the IDF build runs under the *invoking* interpreter, which
# may lack it (newer-platform reprovisioning can drop it) → builds fail with
# `ModuleNotFoundError: No module named 'intelhex'`. Pin it so every pio
# interpreter the workflow uses has it.
intelhex>=2.3,<3
